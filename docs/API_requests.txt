{
	"user_hash": "",
        "notebook_hash" : "", //Optional, limits scope of search to this notebook
        "entry_hash" : "", // Optional, limits scope of search to this entry

}

	"user_hash" : "",
	"notebook_hash" : "",
	"entry_hash" : "",
}
